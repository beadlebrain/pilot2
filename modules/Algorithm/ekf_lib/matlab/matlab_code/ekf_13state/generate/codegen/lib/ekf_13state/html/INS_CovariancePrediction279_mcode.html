1% // *************  CovariancePrediction *************
 2% // Does the prediction step of the Kalman filter for the covariance matrix
 3% // Output, Pnew, overwrites P, the input covariance
 4% // Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G'
 5% // Q is the discrete time covariance of process noise
 6% // Q is vector of the diagonal for a square matrix with
 7% // dimensions equal to the number of disturbance noise variables
 8% // The General Method is very inefficient,not taking advantage of the sparse F and G
 9% // The first Method is very specific to this implementation
10% // ************************************************
11function P = INS_CovariancePrediction(F,G,Q,dT,P)
12I=eye(13);
13P=(I+F*dT)*P*(I+F*dT)'+dT^2*G*Q*G';
